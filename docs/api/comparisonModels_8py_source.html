5def evaluate_binary_prob_model(y_true, p_pred, threshold: float = 0.5) -> dict:
6 """Compute common metrics for binary probabilistic predictions.
7
8 Args:
9 y_true: Iterable of 0/1 labels.
10 p_pred: Iterable of predicted probabilities for class 1.
11 threshold: Decision threshold used for accuracy.
12
13 Returns:
14 Dict with keys: log_loss, brier, accuracy, auc.
15 AUC is NaN if only one class is present.
16 """
17 y_true = np.asarray(y_true).astype(int)
18 p_pred = np.asarray(p_pred).astype(float)
19 p_pred = np.clip(p_pred, 1e-15, 1 - 1e-15)
20
21 metrics = {
22 "log_loss": float(log_loss(y_true, p_pred, labels=[0, 1])),
23 "brier": float(brier_score_loss(y_true, p_pred)),
24 "accuracy": float(accuracy_score(y_true, (p_pred >= threshold).astype(int))),
25 }
26
27 if len(np.unique(y_true)) == 2:
28 metrics["auc"] = float(roc_auc_score(y_true, p_pred))
29 else:
30 metrics["auc"] = float("nan")
31
32 return metrics

35def bootstrap_ci_diff(
36 y_true,
37 p_a,
38 p_b,
39 metric_fn,
40 n_boot: int = 2000,
41 seed: int = 42,
42) -> tuple[float, float, float]:
43 """Bootstrap a confidence interval of a paired metric difference.
44
45 Returns:
46 (ci_low, ci_high, mean_diff) for (metric_a - metric_b).
47
48 Notes:
49 If a re-sample makes the metric undefined (e.g., only a single class for AUC/logloss etc.),
50 that sample is skipped. If too many samples are skipped, CI might be wide.
51 """
52 rng = np.random.default_rng(seed)
53 y_true = np.asarray(y_true)
54 p_a = np.asarray(p_a)
55 p_b = np.asarray(p_b)
56
57 n = len(y_true)
58 diffs: list[float] = []
59 for _ in range(int(n_boot)):
60 idx = rng.integers(0, n, size=n)
61 try:
62 diffs.append(float(metric_fn(y_true[idx], p_a[idx]) - metric_fn(y_true[idx], p_b[idx])))
63 except ValueError:
64 # e.g. bootstrap resample with a single class for AUC/logloss etc.
65 continue
66
67 if not diffs:
68 raise ValueError("Bootstrap produced no valid samples (metric undefined on all resamples).")
69
70 diffs_arr = np.asarray(diffs, dtype=float)
71 return (
72 float(np.quantile(diffs_arr, 0.025)),
73 float(np.quantile(diffs_arr, 0.975)),
74 float(diffs_arr.mean()),
75 )
