41def best_name_match(query: str, candidates: list[str]) -> str | None:
42 q = norm(query)
43 for c in candidates:
44 if norm(c) == q:
45 return c
46 m = get_close_matches(query, candidates, n=1, cutoff=0.75)

57def resolve_player_id(hub: DataHub, name_query: str):
58 players = hub.load_players()

238 def _expected_feature_names(estimator):
239 """Best-effort extraction of expected feature names for sklearn estimators/pipelines."""
240 # Direct estimator
241 names = getattr(estimator, "feature_names_in_", None)
242 if names is not None:
243 return list(names)
244
245 # Pipelines: try final estimator
246 if hasattr(estimator, "steps"):
247 try:
248 final_est = estimator.steps[-1][1]
249 names = getattr(final_est, "feature_names_in_", None)
250 if names is not None:
251 return list(names)
252 except Exception:
253 pass
254
255 # Otherwise try any step
256 for _, step in getattr(estimator, "steps", []):
257 names = getattr(step, "feature_names_in_", None)
258 if names is not None:
259 return list(names)
260
261 return None
