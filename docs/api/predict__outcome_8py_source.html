74def predict_outcome(req: PredictRequest) -> PredictResult:
75 bundle = joblib.load(req.model_path)
76 model = bundle["model"]
77 feature_cols: list[str] = bundle["features"]
78 train_end_year = bundle.get("train_end_year")

87 p1_id, p1_resolved = _resolve_player(players_df, req.p1_name)
88 p2_id, p2_resolved = _resolve_player(players_df, req.p2_name)
89
90 recent_feature_cols = {"recent_form_missing_A", "recent_form_missing_B", *RECENT_FEATURE_MAP.values()}

92
93 matches_df = None
94 if needs_recent_form or req.best_of is None:
95 matches_df = _load_matches_for_prediction(hub, target_date, {p1_id, p2_id})
96
97 best_of_value, best_of_source = _resolve_best_of(
98 req,
99 matches_df if matches_df is not None else pd.DataFrame(),
100 p1_resolved,
101 p2_resolved,
102 target_date,
103 )
104 req_features = replace(req, best_of=best_of_value)
105
106 base_row = _build_base_feature_row(
107 p1_id,
108 p1_resolved,
109 p2_id,
110 p2_resolved,
111 target_date,
112 req_features,
113 rankings_df,
114 lookup,
115 )
116 dataset = pd.DataFrame([base_row])
117 dataset = add_one_hot_features(dataset)
118 dataset = add_derived_features(dataset)
119
120 if needs_recent_form and matches_df is not None:
121 matches_history = matches_df[matches_df["tourney_date"] < target_date]
122 dataset = add_recent_form_features(matches_history, dataset)
123
124 dataset = dataset.fillna(np.nan)
125 row_values = dataset.iloc[0]
126 A_name = row_values["A_name"]
127 B_name = row_values["B_name"]
128
129 for col in feature_cols:
130 if col not in dataset.columns:
131 dataset[col] = 0.0
132
133 X = dataset[feature_cols]
134 probs = model.predict_proba(X)[0]
135 p_A_win = float(probs[1])
136 p_B_win = 1.0 - p_A_win
137
138 canonical_A_is_p1 = normalize_name(A_name) == normalize_name(p1_resolved)
139 if canonical_A_is_p1:
140 p_p1_win = p_A_win
141 p_p2_win = p_B_win
142 else:
143 p_p1_win = p_B_win
144 p_p2_win = p_A_win
145
146 meta = {
147 "model_type": bundle.get("model_type"),
148 "train_end_year": train_end_year,
149 "val_end_year": val_end_year,
150 "feature_count": len(feature_cols),
151 "target_date": target_date.date(),
152 "surface": row_values.get("surface"),
153 "round": row_values.get("round"),
154 "best_of": row_values.get("best_of"),
155 "best_of_source": best_of_source,
156 "p1_resolved": p1_resolved,
157 "p2_resolved": p2_resolved,
158 }
159
160 return PredictResult(
161 A_name=A_name,
162 B_name=B_name,
163 p_A_win=p_A_win,
164 p_B_win=p_B_win,
165 p_p1_win=p_p1_win,
166 p_p2_win=p_p2_win,
167 canonical_A_is_p1=canonical_A_is_p1,
168 meta=meta,
169 )

180def _resolve_player(players_df: pd.DataFrame, query: str) -> tuple[int, str]:
181 if "full_name" not in players_df.columns:
182 raise ValueError("Le fichier joueurs ne contient pas la colonne full_name.")
183 candidates = players_df["full_name"].astype(str)
184 normalized = candidates.map(_norm)
185 target = _norm(query)
186 match_df = players_df[normalized == target]
187 if not match_df.empty:
188 row = match_df.iloc[0]
189 else:
190 best = get_close_matches(query, candidates.tolist(), n=1, cutoff=0.7)
191 if not best:
192 raise ValueError(f"Joueur introuvable: {query}")
193 row = players_df[candidates == best[0]].iloc[0]
194 player_id = row.get("player_id")
195 if pd.isna(player_id):
196 raise ValueError(f"Identifiant joueur manquant pour {row['full_name']}")
197 return int(player_id), str(row["full_name"])

200def _filter_matches_for_players(matches_df: pd.DataFrame, player_ids: set[int]) -> pd.DataFrame:
201 if matches_df.empty or not player_ids:
202 return matches_df
203 if "winner_id" not in matches_df.columns or "loser_id" not in matches_df.columns:
204 return matches_df
205 filtered = matches_df.copy()
206 filtered["winner_id"] = pd.to_numeric(filtered["winner_id"], errors="coerce")
207 filtered["loser_id"] = pd.to_numeric(filtered["loser_id"], errors="coerce")
208 mask = filtered["winner_id"].isin(player_ids) | filtered["loser_id"].isin(player_ids)
209 return filtered[mask].reset_index(drop=True)

241def _default_best_of(req: PredictRequest) -> int:
242 level = str(req.tourney_level or "").strip().upper()
243 if level == "G":
244 return 5
245 name = str(req.tourney_name or "").strip().casefold()
246 for slam in ("wimbledon", "roland garros", "us open", "australian open"):

252def _infer_best_of_from_matches(
253 matches_df: pd.DataFrame,
254 p1_name: str,
255 p2_name: str,
256 target_date: pd.Timestamp,
257 surface: Optional[str],
258 round_value: Optional[str],
259) -> Optional[int]:
260 if matches_df.empty or pd.isna(target_date):
261 return None
262 if "tourney_date" not in matches_df.columns:
263 return None
264 if "winner_name" not in matches_df.columns or "loser_name" not in matches_df.columns:
265 return None
266
267 target_date = pd.to_datetime(target_date).normalize()
268 start = target_date - pd.Timedelta(days=21)
269 end = target_date + pd.Timedelta(days=1)
270 candidates = matches_df[matches_df["tourney_date"].between(start, end)]
271 if candidates.empty:
272 return None
273
274 p1_key = normalize_name(p1_name)
275 p2_key = normalize_name(p2_name)
276 winner_key = candidates["winner_name"].astype(str).map(normalize_name)
277 loser_key = candidates["loser_name"].astype(str).map(normalize_name)
278 player_mask = ((winner_key == p1_key) & (loser_key == p2_key)) | ((winner_key == p2_key) & (loser_key == p1_key))
279 candidates = candidates[player_mask]
280 if candidates.empty:
281 return None
282
283 if round_value and "round" in candidates.columns:
284 round_norm = str(round_value).strip().upper()
285 round_filtered = candidates[candidates["round"].astype(str).str.strip().str.upper() == round_norm]
286 if not round_filtered.empty:
287 candidates = round_filtered
288
289 if surface and "surface" in candidates.columns:
290 surface_norm = str(surface).strip().title()
291 surface_filtered = candidates[candidates["surface"].astype(str).str.strip().str.title() == surface_norm]
292 if not surface_filtered.empty:
293 candidates = surface_filtered
294
295 candidates = candidates.copy()
296 candidates["__date_delta"] = (candidates["tourney_date"] - target_date).abs()
297 candidates = candidates.sort_values(["__date_delta"])
298 return _safe_int(candidates.iloc[0].get("best_of"))
